let valor = 0;
function ligar_desligar(){
    var cor = document.getElementById("botao_checar").value;

    if(cor == true){
        document.getElementById("barrademenu").style.backgroundColor = '#FFFFFF';
        document.getElementById("fim").style.backgroundColor = '#FFFFFF';
        document.body.style.backgroundColor = '#F0F0F0';
        document.body.style.color = '#4f4f4f';

        document.getElementById("tudo").style.borderColor = '#0D0D0D';
        
        document.getElementById("sharkel1").style.backgroundColor = '#FFFFFF';
        document.getElementById("sharkel2").style.backgroundColor = '#FFFFFF';
        document.getElementById("sharkel3").style.backgroundColor = '#FFFFFF';
/*        document.getElementById("sharkel4").style.backgroundColor = '#FFFFFF';
        document.getElementById("sharkel5").style.backgroundColor = '#FFFFFF';
        document.getElementById("sharkel6").style.backgroundColor = '#FFFFFF';
        document.getElementById("sharkel7").style.backgroundColor = '#FFFFFF';
*/

        cor = document.getElementById("botao_checar").value=0;
    }else{

        document.getElementById("barrademenu").style.backgroundColor = '#0D0D0D';
        document.getElementById("fim").style.backgroundColor = '#0D0D0D';
        document.body.style.backgroundColor = '#1F1F1F';
        document.body.style.color = '#CCC';
        
        document.getElementById("tudo").style.borderColor = '#F0F0F0';

        document.getElementById("sharkel1").style.backgroundColor = '#0D0D0D';
        document.getElementById("sharkel2").style.backgroundColor = '#0D0D0D';
        document.getElementById("sharkel3").style.backgroundColor = '#0D0D0D';
/*        document.getElementById("sharkel4").style.backgroundColor = '#0D0D0D';
        document.getElementById("sharkel5").style.backgroundColor = '#0D0D0D';
        document.getElementById("sharkel6").style.backgroundColor = '#0D0D0D';
        document.getElementById("sharkel7").style.backgroundColor = '#0D0D0D';
*/

	    cor = document.getElementById("botao_checar").value=1;
        }

    localStorage.setItem('tema', cor);
    




    var imagem = document.getElementById("botao_checar");
    
    if(valor == 0) {
        if(imagem.style.getPropertyValue('transform').valueOf() != "rotate(360deg)") {
            imagem.style.transform = 'rotate(360deg)';
        }else if(imagem.style.getPropertyValue('transform').valueOf() == "rotate(360deg)"){
            imagem.style.transform = 'rotate(0deg)';
        }
        imagem.style.opacity = "0";
    }else {
        imagem.style.opacity = "0";
        if(imagem.style.getPropertyValue('transform').valueOf() != "rotate(360deg)") {
            imagem.style.transform = 'rotate(360deg)';
        }else if(imagem.style.getPropertyValue('transform').valueOf() == "rotate(360deg)"){
            imagem.style.transform = 'rotate(0deg)';
        }
    }

    const timer = setInterval(() => {
        if(valor == 0) {
            imagem.src = "../dagroins/darktutu.png";
            imagem.style.opacity = "1";
            valor = 1;
            localStorage.setItem('tubarao', valor);
            clearInterval(timer);
        }else{
            imagem.src = "../dagroins/whitetutu.png";
            imagem.style.opacity = "1";
            valor = 0;
            localStorage.setItem('tubarao', valor);
            clearInterval(timer);
        }
    }, 150);

}
